program task_9_27;
var
    i, j, n, a, x: integer;
    fl: boolean;
begin
    fl := true;
    i := 1;
    while (i <> n) and fl do
        begin
            j := i + 1;
            while (j <> n) and fl do
                begin
                    if a <> a then
                        fl := false;
                    j := j + 1;
                end;
            i := i + 1;
        end;
    if fl then
        write(x)
    else
        write(x);
end.
